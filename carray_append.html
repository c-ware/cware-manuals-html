NAME

carray_append - add data to the end of an array

SYNOPSIS

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/carray/carray.h>

#define carray_append(array, value, namespace);

DESCRIPTION

The carray_append macro will add data to the end of an array. If the array is full, then the array will be resized using the CWARE_LIBCARRAY_RESIZE macro function. This can only occur in heap mode, however.

This macro will print an error message to the stderr, and exit the program when any of the following conditions are met:

•

The array is full in stack mode

array will be the array to append to
value will be the value to append
namespace will be the namespace of the array settings to use

EXAMPLES

/*
* This example showcases how to use the carray_append macro.
*/

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/carray/carray.h>

struct StringArray {
int length;
int capacity;
char **contents;
};

int main(void) {
struct StringArray *array = carray_init(array, STRING_ARRAY);

/* Append some data to the array */
carray_append(array, "foo", STRING_ARRAY);
carray_append(array, "bar", STRING_ARRAY);
carray_append(array, "baz", STRING_ARRAY);

/* Release the allocated array */
carray_append(array, STRING_ARRAY);
/* Release the array from memory */
carray_free(array, STRING_ARRAY);

return EXIT_SUCCESS;
}
