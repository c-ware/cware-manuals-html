NAME

chash_lookup - retrieve the value assigned to a key

SYNOPSIS

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/chash/chash.h>

#define chash_lookup(hashtable, _key, storage, namespace);

DESCRIPTION

The chash_lookup macro will retrieve the value assigned to a key, and write it into the variable storage. If the key cannot be found in the hashtable, an error is raised. If you would like to simply detect if a key exists in a hashtable without raising an error if it is not, use chash_contains(cware).

This macro will print an error message to the stderr, and exit the program when any of the following conditions are met:

•

The buffer is NULL

•

The key does not exist in the hashtable

hashtable will be the hashtable to lookup in
_key will be the key to lookup
storage will be the location to write the value to
namespace will be the namespace of the hashtable settings to use

EXAMPLES

/*
* This example showcases how to use the chash_lookup macro.
*/

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/chash/chash.h>

struct StringBucket {
char *key;
int value;
int state;
};

struct StringHashtable {
int length;
int capacity;
struct StringBucket *buckets;
};

int main(void) {
char *value = NULL;
struct StringHashtable *hashtable = chash_init(hashtable, STRING_TABLE);

/* Add some keys */
chash_assign(hashtable, "foo", 1, STRING_TABLE);
chash_assign(hashtable, "bar", 2, STRING_TABLE);
chash_assign(hashtable, "baz", 3, STRING_TABLE);

/* Retrieve and display their values */
value = chash_lookup(hashtable, "foo", value, STRING_TABLE);
printf("%s\n", value);
value = chash_lookup(hashtable, "bar", value, STRING_TABLE);
printf("%s\n", value);
value = chash_lookup(hashtable, "baz", value, STRING_TABLE);
printf("%s\n", value);

/* Free the hashtable */
chash_free(hashtable, STRING_TABLE);

return EXIT_SUCCESS;
}

CONFORMING TO

This macro conforms to C89.
