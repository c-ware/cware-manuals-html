NAME

carray_remove - remove the first occurrence of a value from an array

SYNOPSIS

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/carray/carray.h>

#define carray_remove(array, value, namespace);

DESCRIPTION

The carray_remove macro will remove the first occurrence of a value from an array, and do nothing more. If the value is not found within the array, then the program will exit and display an error message.

This macro will print an error message to the stderr, and exit the program when any of the following conditions are met:

•

The value could not be found

array will be the array to remove from
value will be the value to remove
namespace will be the namespace of the array settings to use

EXAMPLES

/*
* This example showcases how to use the carray_remove macro.
*/

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/carray/carray.h>

struct StringArray {
int length;
int capacity;
char **contents;
};

int main(void) {
struct StringArray *array = carray_init(array, STRING_ARRAY);

/* Append some data to the array */
carray_append(array, "foo", STRING_ARRAY);
carray_append(array, "bar", STRING_ARRAY);
carray_append(array, "baz", STRING_ARRAY);

/* Remove the first occurrence of "bar" */
carray_remove(array, "bar", STRING_ARRAY);

/* Remember to release the array after! */
carray_free(array, STRING_ARRAY);

return EXIT_SUCCESS;
}
