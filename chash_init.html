NAME

chash_init - initialise a new hashtable on the heap

SYNOPSIS

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/chash/chash.h>

#define chash_init(hashtable, namespace);

DESCRIPTION

The chash_init macro will initialise a new hashtable on the heap, and write it into the variable hashtable. The hashtable produced by this macro has to be released when the programmer decides it is most fitting to delete it, which can be done with chash_free(cware).

This macro has no error conditions.

hashtable will be the variable to initialise the hashtable into
namespace will be the namespace of the hashtable settings to use

EXAMPLES

/*
* This example showcases how to use the chash_init macro.
*/

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/chash/chash.h>

struct StringBucket {
char *key;
int value;
int state;
};

struct StringHashtable {
int length;
int capacity;
struct StringBucket *buckets;
};

int main(void) {
struct StringHashtable *hashtable = chash_init(hashtable, STRING_TABLE);

/* Perform operations on the hashtable */
chash_assign(hashtable, "foo", 1);
chash_assign(hashtable, "bar", 2);

/* Free the hashtable */
chash_free(hashtable, STRING_TABLE);

return EXIT_SUCCESS;
}

CONFORMING TO

This macro conforms to C89.
