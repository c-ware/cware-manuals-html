NAME

chash_delete - delete a key value pair from the hashtable

SYNOPSIS

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/chash/chash.h>

#define chash_delete(hashtable, _key, namespace);

DESCRIPTION

The chash_delete macro will delete a key value pair from a hashtable, and mark the spot that they existed inside of as a tombstone.

This macro will print an error message to the stderr, and exit the program when any of the following conditions are met:

•

The buffer is NULL

•

The key does not exist in the hashtable

hashtable will be the hashtable to delete from
_key will be the key to delete the value of
namespace will be the namespace of the hashtable settings to use

EXAMPLES

/*
* This example showcases how to use the chash_delete macro.
*/

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/chash/chash.h>

struct StringBucket {
char *key;
int value;
int state;
};

struct StringHashtable {
int length;
int capacity;
struct StringBucket *buckets;
};

int main(void) {
char *value = NULL;
struct StringHashtable *hashtable = chash_init(hashtable, STRING_TABLE);

/* Add some keys */
chash_assign(hashtable, "foo", 1, STRING_TABLE);
chash_assign(hashtable, "bar", 2, STRING_TABLE);
chash_assign(hashtable, "baz", 3, STRING_TABLE);

/* Remove some keys */
chash_delete(hashtable, "foo", STRING_TABLE);

/* Free the hashtable */
chash_free(hashtable, STRING_TABLE);

return EXIT_SUCCESS;
}

CONFORMING TO

This macro conforms to C89.
