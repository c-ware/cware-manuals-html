NAME

chash_free - release the hashtable and its contents from memory

SYNOPSIS

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/chash/chash.h>

#define chash_free(hashtable, namespace);

DESCRIPTION

The chash_free macro will release the contents of the hashtable from memory, as well as the hashtable itself if the hashtable is in stack mode. It will invoke the HASHTABLE_NAME_FREE_KEY and HASHTABLE_NAME_FREE_VALUE macros on each key value pair that is not marked as empty or a tombstone.

This macro will print an error message to the stderr, and exit the program when any of the following conditions are met:

•

The buffer is NULL

hashtable will be the hashtable to free
namespace will be the namespace of the hashtable settings to use

EXAMPLES

/*
* This example showcases how to use the chash_free macro.
*/

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/chash/chash.h>

struct StringBucket {
char *key;
int value;
int state;
};

struct StringHashtable {
int length;
int capacity;
struct StringBucket *buckets;
};

int main(void) {
int exists = -1;
struct StringHashtable *hashtable = chash_init(hashtable, STRING_TABLE);

/* Add some keys */
chash_assign(hashtable, "foo", 1, STRING_TABLE);
chash_assign(hashtable, "bar", 2, STRING_TABLE);
chash_assign(hashtable, "baz", 3, STRING_TABLE);

/* Free the hashtable */
chash_free(hashtable, STRING_TABLE);

return EXIT_SUCCESS;
}

CONFORMING TO

This macro conforms to C89.
