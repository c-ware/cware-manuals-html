NAME

libpath_rmdir - remove a directory

SYNOPSIS

#include "libpath.h"

int libpath_rmdir(const char *path);

DESCRIPTION

Removes a directory at the path specified. Note that some operating systems may not allow you to remove a directory if it is not empty.

This function will display an error to the stderr, and abort the program if any of following conditions are met.

o path is NULL

path will be the path of the directory to remove

RETURN VALUE

This function will return LIBPATH_SUCCESS on success, and LIBPATH_FAILURE on failure

NOTES

This function has no notes.

EXAMPLES

#include <stdio.h>
#include <errno.h>
#include <stdlib.h>
#include <string.h>

#include "libpath.h"

int main(void) {
if(libpath_rmdir(LIBPATH_ROOT_DIR, "prog_config") == LIBPATH_FAILURE) {
fprintf(stderr, "prog: failed to remove directory (%s)0, strerror(errno));
exit(EXIT_FAILURE);
}

return 0;
}
