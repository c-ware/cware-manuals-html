NAME

carray_init - initialise a new array on the heap

SYNOPSIS

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/carray/carray.h>

#define carray_init(array, namespace);

DESCRIPTION

The carray_init macro initialises a new array on the heap. The resulting array must be released from memory at the end of the program. The array can also grow as more items are inserted into it.

This function has no error conditions.

array will be the array variable to initialise the array into
namespace will be the namespace of the array settings to use

EXAMPLES

/*
* This example showcases how to use the carray_pop macro.
*/

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/carray/carray.h>

struct StringArray {
int length;
int capacity;
char **contents;
};

int main(void) {
struct StringArray *array = carray_init(array, STRING_ARRAY);

/* Remember to release the array after! */
carray_free(array, STRING_ARRAY);

return EXIT_SUCCESS;
}
