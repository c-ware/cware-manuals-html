NAME

carray_is_empty - determine if an array is empty

SYNOPSIS

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/carray/carray.h>

#define carray_is_empty(array);

DESCRIPTION

The carray_is_empty macro will determine if an array is empty. If the array is empty, this will return 1. If it is not empty, this will return 0.

This macro has no error conditions.

array will be the array to check for emptiness

EXAMPLES

/*
* This example showcases how to use the carray_is_empty macro.
*/

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/carray/carray.h>

struct StringArray {
int length;
int capacity;
char **contents;
};

int main(void) {
struct StringArray *array = carray_init(array, STRING_ARRAY);

/* Append some data to the array */
carray_append(array, "foo", STRING_ARRAY);
carray_append(array, "bar", STRING_ARRAY);
carray_append(array, "baz", STRING_ARRAY);

/* Check if the array is empty */
if(carray_is_empty(array) == 1) {
printf("%s", "The array is empty.\n");
} else {
printf("%s", "The array is not empty.\n");
}

/* Release the allocated array */
carray_is_empty(array, STRING_ARRAY);

return EXIT_SUCCESS;
}
