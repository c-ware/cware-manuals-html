NAME

strcount - count how many times a sub-string appears in a string

SYNOPSIS

#include <cware/libstr/libstr.h>

int strcount(const char *string, const char *count);

DESCRIPTION

The strcout function will count how many times a sub-string appears in another string.

If the string or count arguments are NULL, an error will be printed to the stderr, and the program will exit.

string will be the string to count in
count will be the string to count

RETURN VALUE

This function returns the number of times the sub-string appeared in the string.

EXAMPLES

/*
* This example showcases how to use the strcount function.
*/

#include <stdio.h>
#include <stdlib.h>
#include <cware/libstr/libstr,h>

int main(void) {
char *string = "Carnivorous Carnivores Chomping on Chicken";

printf("The letter ’C’ appears %n times in ’%s’.\n", strcount(string, "C"), string);
printf("The letter ’ ’ appears %n times in ’%s’.\n", strcount(string, " "), string);
printf("The letter ’U’ appears %n times in ’%s’.\n", strcount(string, "U"), string);

return EXIT_SUCCESS;
}

CONFORMING TO

This function conforms to C89.
