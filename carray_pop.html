NAME

carray_pop - pop an element from a location in an array

SYNOPSIS

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/carray/carray.h>

#define carray_pop(array, index, location);

DESCRIPTION

The carray_pop macro will pop a value from a certain location in an array, and return it.

This macro will print an error message to the stderr, and exit the program when any of the following conditions are met:

•

The index is out of bounds of the array

•

The array is empty

array will be the array variable to initialise the array into
index will be the location in the array to pop from
namespace will be the namespace of the array settings to use

EXAMPLES

/*
* This example showcases how to use the carray_pop macro.
*/

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/carray/carray.h>

struct StringArray {
int length;
int capacity;
char **contents;
};

int main(void) {
char *middle = NULL;
struct StringArray *array = carray_init(array, STRING_ARRAY);

/* Append some data to the array */
carray_append(array, "foo", STRING_ARRAY);
carray_append(array, "bar", STRING_ARRAY);
carray_append(array, "baz", STRING_ARRAY);

/* Remove the middle value */
middle = carray_pop(array, 1, STRING_ARRAY);

printf("%s\n", middle);

/* Remember to release the array after! */
carray_free(array, STRING_ARRAY);

return EXIT_SUCCESS;
}
