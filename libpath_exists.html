NAME

libpath_exists - determine if a path exists on the file system

SYNOPSIS

#include "libpath.h"

int libpath_exists(const char *path);

DESCRIPTION

Determine if a path exists on the file system. This function is garunteed to work on all types of file system components. Specifically, files AND directories.

This function will display an error to the stderr, and abort the program if any of following conditions are met.

o path is NULL

path will be the path to check

RETURN VALUE

This function will return 1 if the path exists, and 0 if it does not

NOTES

This function has no notes.

EXAMPLES

#include <stdio.h>
#include <errno.h>
#include <stdlib.h>
#include <string.h>

#include "libpath.h"

int main(void) {
if(libpath_exists(LIBPATH_ROOT_DIR, "prog_config") == LIBPATH_FAILURE) {
fprintf(stderr, "%s", "prog: cannot make directory-- already exists0);
exit(EXIT_FAILURE);
}

return 0;
}
