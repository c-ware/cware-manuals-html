NAME

carray_free - release the array and its contents from memory

SYNOPSIS

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/carray/carray.h>

#define carray_free(array, namespace);

DESCRIPTION

The carray_free macro will release the array from memory, along with its contents and its buffer. It makes use of the ARRAY_NAME_FREE macro function to release each value from memory. If there is no need to free the data inside the array, like if the data is on the stack, then that macro should be empty.

This macro has no error conditions.

array will be the array to free
namespace will be the namespace of the array settings to use

EXAMPLES

/*
* This example showcases how to use the carray_free macro.
*/

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/carray/carray.h>

struct StringArray {
int length;
int capacity;
char **contents;
};

int main(void) {
struct StringArray *array = carray_init(array, STRING_ARRAY);

/* Append some data to the array */
carray_append(array, "foo", STRING_ARRAY);
carray_append(array, "bar", STRING_ARRAY);
carray_append(array, "baz", STRING_ARRAY);

/* Release the allocated array */
carray_free(array, STRING_ARRAY);

return EXIT_SUCCESS;
}
