NAME

carray_find - find the first occurrence of a value in an array

SYNOPSIS

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/carray/carray.h>

#define carray_find(array, value, , location, namespace);

DESCRIPTION

The carray_find macro will determine the location (index) of a value in an array, writing its index to the location argument. If the value could not be found, then -1 will be written to the variable.

This macro has no error conditions.

array will be the array to search
value will be the value to find
location will be the location to write the index to
namespace will be the namespace of the array settings to use

EXAMPLES

/*
* This example showcases how to use the carray_find macro.
*/

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/carray/carray.h>

struct StringArray {
int length;
int capacity;
char **contents;
};

int main(void) {
int foo_location = 0;
int bar_location = 0;
int baz_location = 0;
int tuna_location = 0;

struct StringArray *array = carray_init(array, STRING_ARRAY);

/* Append some data to the array */
carray_append(array, "foo", STRING_ARRAY);
carray_append(array, "bar", STRING_ARRAY);
carray_append(array, "baz", STRING_ARRAY);

/* Find the locations of all the elements */
carray_find(array, "foo", foo_location, STRING_ARRAY);
carray_find(array, "bar", bar_location, STRING_ARRAY);
carray_find(array, "baz", baz_location, STRING_ARRAY);
carray_find(array, "tuna", tuna_location, STRING_ARRAY);

/* Display the locations */
printf("foo at index: %i\n", foo_location);
printf("bar at index: %i\n", bar_location);
printf("baz at index: %i\n", baz_location);
printf("tuna at index: %i\n", tuna_location);

/* Remember to release the array after! */
carray_free(array, STRING_ARRAY);

return EXIT_SUCCESS;
}
