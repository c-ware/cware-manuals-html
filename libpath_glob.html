NAME

libpath_glob - extract files and directories from a path based off a pattern

SYNOPSIS

#include "libpath.h"

struct LibpathFiles libpath_glob(const char *path, const char *pattern);

DESCRIPTION

This function will extract all files and subdirectories inside of a directory based off a pattern. The pattern is similar (although fairly limited in comparison to) shell expansions.

Currently, libpath_glob allows the following syntax: *
Wildcard-- match any character until the right-most character, or the NUL byte

This function has no error conditions.

path will be the path to glob in
pattern will be the pattern to glob

RETURN VALUE

This function will return an array of globbed paths

NOTES

This function has no notes.

EXAMPLES

#include <stdio.h>
#include <errno.h>
#include <stdlib.h>
#include <string.h>

#include "libpath.h"

int main(void) {
int index = 0;
struct LibpathFiles globbed_files = libpath_glob("./", "*.*");

// Display all globbed files
for(index = 0; index < globbed_files.length; index++) {
struct LibpathFile file = globbed_files.contents[index];

printf("Globbed file: ’%s’0, file.path);
}

return 0;
}
