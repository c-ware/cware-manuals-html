NAME

carray_insert - insert a value into an array at some location

SYNOPSIS

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/carray/carray.h>

#define carray_insert(array, index, value, namespace);

DESCRIPTION

Inserts a value into an array at some location. The location must be within the bounds of the array. If the array is in heap mode, the array will be resized when the length is equal to the capacity. Otherwise, if the array is full, then an error will occur.

This macro will print an error message to the stderr, and exit the program when any of the following conditions are met:

•

The index is less than zero or greater than the length of the array

•

The array is full in stack mode

array will be the array to insert into
index will be the location in the array to insert into
value will be the value to insert
namespace will be the namespace of the array settings to use

EXAMPLE

/*
* This example showcases how to use the carray_insert macro.
*/

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/carray/carray.h>

struct StringArray {
int length;
int capacity;
char **contents;
};

int main(void) {
struct StringArray *array = carray_init(array, STRING_ARRAY);

/* Insert some values */
carray_insert(array, 0, "foo", STRING_ARRAY);
carray_insert(array, 1, "foo", STRING_ARRAY);
carray_insert(array, 2, "foo", STRING_ARRAY);

/* Release the array */
carray_free(array, STRING_ARRAY);

return EXIT_SUCCESS;
}
