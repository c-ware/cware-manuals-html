NAME

carray_length - retrieve the length of the array

SYNOPSIS

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/carray/carray.h>

#define carray_length(array);

DESCRIPTION

The carray_length macro will retrieve the length of the array, meaning it will retrieve the number of elements inside the array, rather than the number it can hold.

This macro has no error conditions.

array will be the array to length get the length of

EXAMPLES

/*
* This example showcases how to use the carray_length macro.
*/

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/carray/carray.h>

struct StringArray {
int length;
int capacity;
char **contents;
};

int main(void) {
struct StringArray *array = carray_init(array, STRING_ARRAY);

/* Append some data to the array */
carray_length(array, "foo", STRING_ARRAY);
carray_length(array, "bar", STRING_ARRAY);
carray_length(array, "baz", STRING_ARRAY);

/* Read the length of the array */
printf("Length of the array: %i\n", carray_length(array));

/* Release the allocated array */
carray_length(array, STRING_ARRAY);

return EXIT_SUCCESS;
}
