NAME

chash_is_full - determine if a hashtable is full

SYNOPSIS

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/chash/chash.h>

#define chash_is_full(hashtable, namespace);

DESCRIPTION

The chash_is_full macro will determine if a hashtable is full. This is useful when using a stack allocated hashtable, where you will most likely want to know if a hashtable is full or not before adding more elements, since stack hashtables do not dynamically resize themselves.

This macro has no error conditions.

hashtable will be the hashtable to check
namespace will be the namespace of the hashtable settings to use

EXAMPLES

/*
* This example showcases how to use the chash_is_full macro.
*/

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <cware/chash/chash.h>

struct StringBucket {
char *key;
int value;
int state;
};

struct StringHashtable {
int length;
int capacity;
struct StringBucket *buckets;
};

int main(void) {
struct StringBucket buckets[3] = {0};
struct StringHashtable hashtable = chash_init_stack(&hashtable, buckets, 3,
STRING_TABLE);

printf("Hashtable is full: %i\n", chash_is_full(&hashtable, STRING_TABLE);

/* Add some keys */
chash_assign(hashtable, "foo", 1);
chash_assign(hashtable, "bar", 2);
chash_assign(hashtable, "baz", 3);

printf("Hashtable is full: %i\n", chash_is_full(&hashtable, STRING_TABLE);

return EXIT_SUCCESS;
}

CONFORMING TO

This macro conforms to C89.
